function generateLogo(name) {
  const fonts = ["Inter", "Poppins", "Montserrat"];
  const icons = ["bolt", "rocket", "leaf"];
  const layouts = ["top", "left", "center"];

  return {
    text: name,
    font: random(fonts),
    icon: random(icons),
    layout: random(layouts),
  };

import { iconMap, fontMap, colorMap, layouts } from "./assets";

export function generateLogo(input: any) {
  const category = analyze(input.name);

  return {
    name: input.name,
    icon: pick(iconMap[category]),
    font: pick(fontMap[input.style]),
    color: pick(colorMap[input.style]),
    layout: pick(layouts),
  };
}

function pick(arr: string[]) {
  return arr[Math.floor(Math.random() * arr.length)];
}

function analyze(name: string) {
  if (name.toLowerCase().includes("tech")) return "tech";
  if (name.toLowerCase().includes("food")) return "food";
  return "general";
}

import { NextResponse } from "next/server";
import { connectDB } from "@/db/connect";
import Logo from "@/models/Logo";
import { generateLogo } from "@/lib/generator";

export async function POST(req: Request) {
  await connectDB();

  const body = await req.json();

  const logos = Array.from({ length: 6 }).map(() =>
    generateLogo(body)
  );

  await Logo.insertMany(logos);

  return NextResponse.json(logos);
}

export function generateLogo(name: string) {
  const icons = ["bolt", "rocket", "leaf"];
  const fonts = ["Arial", "Poppins", "Inter"];
  const colors = ["#111", "#2563eb", "#dc2626"];

  return {
    name,
    icon: icons[Math.floor(Math.random() * icons.length)],
    font: fonts[Math.floor(Math.random() * fonts.length)],
    color: colors[Math.floor(Math.random() * colors.length)],
  };
}

import { NextResponse } from "next/server";
import { generateLogo } from "@/lib/generator";

export async function POST(req: Request) {
  const body = await req.json();

  const logos = Array.from({ length: 6 }).map(() =>
    generateLogo(body.name)
  );

  return NextResponse.json(logos);
}

"use client";

import { useState } from "react";
import LogoView from "@/components/LogoView";

export default function Generate() {
  const [name, setName] = useState("");
  const [industry, setIndustry] = useState("technology");
  const [description, setDescription] = useState("");
  const [styles, setStyles] = useState<string[]>([]);
  const [logos, setLogos] = useState<any[]>([]);

  const styleOptions = [
    "Modern",
    "Minimal",
    "Bold",
    "Elegant",
    "Playful",
    "Friendly",
  ];

  function toggleStyle(style: string) {
    if (styles.includes(style)) {
      setStyles(styles.filter((s) => s !== style));
    } else {
      setStyles([...styles, style]);
    }
  }

  async function handleGenerate() {
    const res = await fetch("/api/generate", {
      method: "POST",
      body: JSON.stringify({
        name,
        industry,
        description,
        styles,
      }),
    });

    const data = await res.json();
    setLogos(data);
  }

  return (
    <div className="p-10 max-w-5xl mx-auto">

      <h1 className="text-2xl font-bold mb-6">
        Tell us about your brand
      </h1>

      {/* FORM */}
      <div className="space-y-6 mb-10">

        {/* Brand Name */}
        <div>
          <label className="block mb-1 font-medium">Brand Name</label>
          <input
            type="text"
            placeholder="e.g. TechNova"
            value={name}
            onChange={(e) => setName(e.target.value)}
            className="w-full border p-3 rounded"
          />
        </div>

        {/* Industry */}
        <div>
          <label className="block mb-1 font-medium">Industry</label>
          <select
            value={industry}
            onChange={(e) => setIndustry(e.target.value)}
            className="w-full border p-3 rounded"
          >
            <option value="technology">Technology</option>
            <option value="food">Food</option>
            <option value="fashion">Fashion</option>
            <option value="finance">Finance</option>
          </select>
        </div>

        {/* Description */}
        <div>
          <label className="block mb-1 font-medium">
            What does your brand do? (optional)
          </label>
          <textarea
            placeholder="Briefly describe your product or mission..."
            value={description}
            onChange={(e) => setDescription(e.target.value)}
            className="w-full border p-3 rounded"
          />
        </div>

        {/* Style Selection */}
        <div>
          <label className="block mb-2 font-medium">
            Brand Personality
          </label>

          <div className="grid grid-cols-3 gap-3">
            {styleOptions.map((style) => (
              <button
                key={style}
                onClick={() => toggleStyle(style)}
                className={`border p-3 rounded ${
                  styles.includes(style)
                    ? "bg-black text-white"
                    : ""
                }`}
              >
                {style}
              </button>
            ))}
          </div>
        </div>

        {/* Generate Button */}
        <button
          onClick={handleGenerate}
          className="bg-black text-white px-6 py-3 rounded w-full"
        >
          Generate Logos
        </button>

      </div>

      {/* RESULTS */}
      <div className="grid grid-cols-3 gap-4">
        {logos.map((logo, i) => (
          <div key={i} className="border p-4 rounded">
            <LogoView logo={logo} />
          </div>
        ))}
      </div>

    </div>
  );
}

import { iconMap, fontMap, colorMap } from "./assets";

export function generateLogo(input: any) {
  const { name, industry, styles } = input;

  const category = detectCategory(name, industry);

  const icon = pick(iconMap[category] || iconMap["general"]);
  const font = pickFromStyles(fontMap, styles);
  const color = pickFromStyles(colorMap, styles);

  return {
    name,
    icon,
    font,
    color,
  };
}

function pick(arr: string[]) {
  return arr[Math.floor(Math.random() * arr.length)];
}

function pickFromStyles(map: any, styles: string[]) {
  if (!styles || styles.length === 0) {
    return pick(Object.values(map).flat());
  }

  const all: string[] = [];

  styles.forEach((style) => {
    if (map[style]) {
      all.push(...map[style]);
    }
  });

  return pick(all.length ? all : Object.values(map).flat());
}

function detectCategory(name: string, industry: string) {
  const lower = name.toLowerCase();

  if (lower.includes("tech")) return "technology";
  if (lower.includes("food")) return "food";

  return industry || "general";
}

import { iconMap, fontMap, colorMap, layouts } from "./assets";

type Input = {
  name: string;
  industry: string;
  styles: string[];
};

// ===== MAIN FUNCTION =====
export function generateLogo(input: Input) {
  const { name, industry, styles } = input;

  const category = detectCategory(name, industry);

  const icon = pick(iconMap[category] || iconMap["general"]);
  const font = pickFromStyles(fontMap, styles);
  const color = pickFromStyles(colorMap, styles);
  const layout = pick(layouts);

  return {
    name,
    icon,
    font,
    color,
    layout,
  };
}

// ===== PICK RANDOM ITEM =====
function pick(arr: string[]) {
  return arr[Math.floor(Math.random() * arr.length)];
}

// ===== PICK BASED ON STYLE =====
function pickFromStyles(
  map: Record<string, string[]>,
  styles: string[]
) {
  if (!styles || styles.length === 0) {
    return pick(Object.values(map).flat());
  }

  const results: string[] = [];

  styles.forEach((style) => {
    if (map[style]) {
      results.push(...map[style]);
    }
  });

  // اگر چیزی پیدا نشد fallback
  if (results.length === 0) {
    return pick(Object.values(map).flat());
  }

  return pick(results);
}

// ===== CATEGORY DETECTION =====
function detectCategory(name: string, industry: string) {
  const lower = name.toLowerCase();

  if (lower.includes("tech")) return "technology";
  if (lower.includes("food")) return "food";
  if (lower.includes("bank") || lower.includes("finance"))
    return "finance";
  if (lower.includes("fashion") || lower.includes("style"))
    return "fashion";

  return industry || "general";
}

    return pick(Object.values(map).flat());
  }

  return pick(results);
}

function detectCategory(name: string, industry: string) {
  const lower = name.toLowerCase();

  if (lower.includes("tech")) return "technology";
  if (lower.includes("food")) return "food";
  if (lower.includes("bank") || lower.includes("finance"))
    return "finance";
  if (lower.includes("fashion") || lower.includes("style"))
    return "fashion";

  return industry || "general";
}

import { iconMap, fontMap, colorMap, layouts } from "./assets";

type Input = {
  name: string;
  industry: string;
  styles: string[];
};

export function generateLogo(input: Input) {
  const { name, industry, styles } = input;

  const category = detectCategory(name, industry);

  const icon = pick(iconMap[category] || iconMap["general"]);
  const font = pickFromStyles(fontMap, styles);
  const color = pickFromStyles(colorMap, styles);

  // اگر Minimal انتخاب شده → متن-only
  const layout =
    styles?.includes("Minimal") ? "textOnly" : pick(layouts);

  return {
    name,
    icon,
    font,
    color,
    layout,
  };
}

// ---------- helpers ----------
function pick(arr: string[]) {
  return arr[Math.floor(Math.random() * arr.length)];
}

function pickFromStyles(
  map: Record<string, string[]>,
  styles: string[]
) {
  if (!styles || styles.length === 0) {
    return pick(Object.values(map).flat());
  }

  const pool: string[] = [];
  styles.forEach((s) => map[s] && pool.push(...map[s]));

  return pick(pool.length ? pool : Object.values(map).flat());
}

function detectCategory(name: string, industry: string) {
  const n = (name || "").toLowerCase();

  if (n.includes("tech")) return "technology";
  if (n.includes("food") || n.includes("cafe") || n.includes("cook"))
    return "food";
  if (n.includes("bank") || n.includes("fin"))
    return "finance";
  if (n.includes("fashion") || n.includes("style"))
    return "fashion";

  return industry || "general";
}

import * as Icons from "lucide-react";

function toPascalCase(kebab: string) {
  return kebab
    .split("-")
    .map((p) => p.charAt(0).toUpperCase() + p.slice(1))
    .join("");
}

function Icon({ name, color, size = 50 }: any) {
  const key = toPascalCase(name);
  const Comp = (Icons as any)[key] || Icons.Star;
  return <Comp size={size} style={{ color }} />;
}

export default function LogoView({ logo }: any) {
  const { layout, icon, color, name, font } = logo;

  // ===== ICON TOP =====
  if (layout === "iconTop") {
    return (
      <div className="flex flex-col items-center gap-2">
        <Icon name={icon} color={color} />
        <p style={{ fontFamily: font }}>{name}</p>
      </div>
    );
  }

  // ===== ICON LEFT =====
  if (layout === "iconLeft") {
    return (
      <div className="flex items-center gap-3">
        <Icon name={icon} color={color} />
        <p style={{ fontFamily: font }}>{name}</p>
      </div>
    );
  }

  // ===== BADGE =====
  if (layout === "badge") {
    return (
      <div className="flex flex-col items-center gap-2">
        <div
          className="w-16 h-16 rounded-full flex items-center justify-center"
          style={{ background: color }}
        >
          <Icon name={icon} color="#fff" size={28} />
        </div>
        <p style={{ fontFamily: font }}>{name}</p>
      </div>
    );
  }

  // ===== TEXT ONLY =====
  if (layout === "textOnly") {
    return (
      <div className="flex items-center justify-center h-full">
        <p
          style={{
            fontFamily: font,
            color: color,
            fontSize: "22px",
            fontWeight: 700,
          }}
        >
          {name}
        </p>
      </div>
    );
  }

  // ===== ICON BIG =====
  if (layout === "iconBig") {
    return (
      <div className="flex flex-col items-center gap-2">
        <Icon name={icon} color={color} size={70} />
        <p style={{ fontFamily: font, fontSize: 14 }}>{name}</p>
      </div>
    );
  }

  // fallback
  return (
    <div className="flex flex-col items-center">
      <Icon name={icon} color={color} />

export default function LogoView({ logo }: any) {
  return (
    <svg width="250" height="200" viewBox="0 0 250 200">
      
      {/* Background Gradient */}
      <defs>
        <linearGradient id={`grad-${logo.name}`} x1="0" y1="0" x2="1" y2="1">
          <stop offset="0%" stopColor={logo.bg[0]} />
          <stop offset="100%" stopColor={logo.bg[1]} />
        </linearGradient>
      </defs>

      <rect
        width="250"
        height="200"
        rx="20"
        fill={`url(#grad-${logo.name})`}
      />

      {/* Layout Switch */}
      {logo.layout === "top" && (
        <>
          {/* Icon */}
          <g transform="translate(125,60) scale(1.5)">
            {getIcon(logo.icon, logo.color)}
          </g>

          {/* Text */}
          <text
            x="50%"
            y="140"
            textAnchor="middle"
            fontSize="26"
            fontFamily={logo.font}
            fill="#fff"
            fontWeight="600"
          >
            {logo.name}
          </text>
        </>
      )}

      {logo.layout === "left" && (
        <>
          <g transform="translate(60,100) scale(1.2)">
            {getIcon(logo.icon, logo.color)}
          </g>

          <text
            x="140"
            y="110"
            fontSize="22"
            fontFamily={logo.font}
            fill="#fff"
            fontWeight="600"
          >
            {logo.name}
          </text>
        </>
      )}

      {logo.layout === "text" && (

          x="50%"
          y="50%"
          textAnchor="middle"
          fontSize="32"
          fontFamily={logo.font}
          fill="#fff"
          fontWeight="700"
        >
          {logo.name}
        </text>
      )}

    </svg>
  );
}

/* 🎯 ICON SYSTEM */
function getIcon(name: string, color: string) {
  switch (name) {
    case "bolt":
      return (
        <polygon
          points="0,-20 10,0 0,0 10,20 -10,5 0,5"
          fill={color}
        />
      );

    case "leaf":
      return (
        <ellipse cx="0" cy="0" rx="12" ry="20" fill={color} />
      );

    case "rocket":
      return (
        <polygon
          points="0,-20 10,10 0,5 -10,10"
          fill={color}
        />
      );

    case "star":
      return (
        <polygon
          points="0,-15 4,-5 15,-5 6,2 10,12 0,6 -10,12 -6,2 -15,-5 -4,-5"
          fill={color}
        />
      );

    default:
      return <circle cx="0" cy="0" r="10" fill={color} />;
  }
}

// 🎯 Logo Generator (Looka-like MVP)

export function generateLogo(input: any) {
  const { name, industry, styles } = input;

  const category = analyze(name, industry);

  const icon = pickIcon(category);
  const font = pickFont(styles);
  const color = pickColor(category);
  const bg = pickBackground(category);
  const layout = pickLayout();

  return {
    name,
    icon,
    font,
    color,
    bg,
    layout,
  };
}

// 🧠 Brand Analyzer (هوشمند ساده)
function analyze(name: string, industry: string) {
  const lower = name.toLowerCase();

  if (lower.includes("tech")) return "technology";
  if (lower.includes("food")) return "food";
  if (lower.includes("bank")) return "finance";

  return industry || "general";
}

// 🎯 Icon Selection
function pickIcon(category: string) {
  const map: any = {
    technology: ["bolt", "rocket", "cpu"],
    food: ["leaf", "apple", "coffee"],
    fashion: ["star", "sparkles"],
    finance: ["chart", "shield"],
    general: ["circle"],
  };

  return random(map[category] || map.general);
}

// 🎨 Font Selection
function pickFont(styles: string[]) {
  if (!styles || styles.length === 0) return "Arial";

  if (styles.includes("Elegant")) return "serif";
  if (styles.includes("Bold")) return "Impact";
  if (styles.includes("Playful")) return "Comic Sans MS";
  if (styles.includes("Minimal")) return "sans-serif";

  return "Arial";
}

// 🌈 Color Selection
function pickColor(category: string) {

    general: ["#ffffff"],
  };

  return random(map[category] || map.general);
}

import { useState, useEffect, useRef } from "react";
import LogoView from "@/components/LogoView";
import * as htmlToImage from "html-to-image";

export default function Editor() {
  const [logo, setLogo] = useState<any>(null);
  const [paid, setPaid] = useState(false);
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const saved = localStorage.getItem("selectedLogo");
    if (saved) setLogo(JSON.parse(saved));
  }, []);

  if (!logo) return <p className="p-10">No logo selected</p>;

  async function downloadPNG() {
    if (!ref.current) return;

    const dataUrl = await htmlToImage.toPng(ref.current, {
      backgroundColor: "white",
      pixelRatio: 4, // 👈 کیفیت بالا
    });

    const a = document.createElement("a");
    a.download = ${logo.name}.png;
    a.href = dataUrl;
    a.click();
  }

  return (
    <div className="p-10 max-w-4xl mx-auto">
      <h1 className="text-2xl font-bold mb-6">Customize your logo</h1>

      {/* PREVIEW */}
      <div
        ref={ref}
        className="border p-10 mb-8 flex justify-center bg-white"
      >
        <LogoView logo={logo} />
      </div>

      {/* CONTROLS */}
      <div className="space-y-6">
        <div>
          <p className="mb-2 font-medium">Color</p>
          <input
            type="color"
            value={logo.color}
            onChange={(e) =>
              setLogo({ ...logo, color: e.target.value })
            }
          />
        </div>

        <div>
          <p className="mb-2 font-medium">Font</p>
          <select
            value={logo.font}
            onChange={(e) =>
              setLogo({ ...logo, font: e.target.value })
            }
            className="border p-2"
          >
            <option>Inter</option>
            <option>Poppins</option>
          </select>
        </div>

        <div>
          <p className="mb-2 font-medium">Layout</p>
          <select
            value={logo.layout}
            onChange={(e) =>
              setLogo({ ...logo, layout: e.target.value })
            }
            className="border p-2"
          >
            <option value="iconTop">Icon Top</option>
            <option value="iconLeft">Icon Left</option>
            <option value="badge">Badge</option>
            <option value="textOnly">Text Only</option>
            <option value="iconBig">Icon Big</option>
          </select>
        </div>

        {/* PAYWALL */}
        {paid ? (
          <button
            onClick={downloadPNG}
            className="bg-green-600 text-white px-6 py-3 rounded w-full"
          >
            Download PNG
          </button>
        ) : (
          <button
            onClick={() => setPaid(true)}
            className="bg-black text-white px-6 py-3 rounded w-full"
          >
            Unlock Download ($5)
          </button>
        )}
      </div>
    </div>
  );
}

import * as htmlToImage from "html-to-image";

export default function Editor() {
  const [logo, setLogo] = useState<any>(null);
  const [paid, setPaid] = useState(false);
  const ref = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const saved = localStorage.getItem("selectedLogo");
    if (saved) setLogo(JSON.parse(saved));
  }, []);

  if (!logo) return <p className="p-10">No logo selected</p>;

  async function downloadPNG() {
    if (!ref.current) return;

    const dataUrl = await htmlToImage.toPng(ref.current, {
      backgroundColor: "white",
      pixelRatio: 4,
    });

    const a = document.createElement("a");
    a.download = `${logo.name}.png`;
    a.href = dataUrl;
    a.click();
  }

  return (
    <div className="p-10 max-w-4xl mx-auto">
      <h1 className="text-2xl font-bold mb-6">Customize your logo</h1>

      {/* PREVIEW */}
      <div
        ref={ref}
        className="border p-10 mb-8 flex justify-center bg-white rounded-xl shadow"
      >
        <LogoView logo={logo} />
      </div>

      {/* CONTROLS */}
      <div className="space-y-6">

        {/* 🎨 COLOR */}
        <div>
          <p className="mb-2 font-medium">Icon Color</p>
          <input
            type="color"
            value={logo.color}
            onChange={(e) =>
              setLogo({ ...logo, color: e.target.value })
            }
          />
        </div>

        {/* 🌈 BACKGROUND GRADIENT */}
        <div>
          <p className="mb-2 font-medium">Background Gradient</p>
          <div className="flex gap-2">
            <input
              type="color"
              value={logo.bg[0]}
              onChange={(e) =>
                setLogo({
                  ...logo,
                  bg: [e.target.value, logo.bg[1]],
                })
              }
            />
            <input
              type="color"
              value={logo.bg[1]}
              onChange={(e) =>
                setLogo({
                  ...logo,
                  bg: [logo.bg[0], e.target.value],
                })
              }
            />
          </div>
        </div>

        {/* 🔤 FONT */}
        <div>
          <p className="mb-2 font-medium">Font</p>
          <select
            value={logo.font}
            onChange={(e) =>
              setLogo({ ...logo, font: e.target.value })
            }
            className="border p-2 rounded"
          >
            <option value="Inter">Inter</option>
            <option value="Poppins">Poppins</option>
            <option value="serif">Serif</option>
          </select>
        </div>

        {/* 📐 LAYOUT */}
        <div>
          <p className="mb-2 font-medium">Layout</p>
          <select
            value={logo.layout}
            onChange={(e) =>
              setLogo({ ...logo, layout: e.target.value })
            }
            className="border p-2 rounded"
          >
            <option value="top">Icon Top</option>
            <option value="left">Icon Left</option>
            <option value="text">Text Only</option>
          </select>
        </div>

        {/* 💰 PAYWALL */}
        {paid ? (
          <button
            onClick={downloadPNG}
            className="bg-green-600 text-white px-6 py-3 rounded w-full hover:bg-green-700"
          >
            Download PNG
          </button>
        ) : (
          <button
            onClick={() => setPaid(true)}
            className="bg-black text-white px-6 py-3 rounded w-full hover:opacity-90"
          >
            Unlock Download ($5)
          </button>
        )}
      </div>
    </div>
  );
}

import * as Icons from "lucide-react";

export default function LogoView({ logo }: any) {
  return (
    <svg width="250" height="200" viewBox="0 0 250 200">
      {/* Gradient */}
      <defs>
        <linearGradient id={`grad-${logo.name}`} x1="0" y1="0" x2="1" y2="1">
          <stop offset="0%" stopColor={logo.bg[0]} />
          <stop offset="100%" stopColor={logo.bg[1]} />
        </linearGradient>
      </defs>

      <rect
        width="250"
        height="200"
        rx="20"
        fill={`url(#grad-${logo.name})`}
      />

      {/* Layouts */}
      {logo.layout === "top" && (
        <>
          <g transform="translate(125,70) scale(1.3)">
            <IconRenderer name={logo.icon} color={logo.color} />
          </g>

          <text
            x="50%"
            y="150"
            textAnchor="middle"
            fontSize="26"
            fontFamily={logo.font}
            fill="#fff"
            fontWeight="600"
          >
            {logo.name}
          </text>
        </>
      )}

      {logo.layout === "left" && (
        <>
          <g transform="translate(70,110) scale(1.2)">
            <IconRenderer name={logo.icon} color={logo.color} />
          </g>

          <text
            x="140"
            y="115"
            fontSize="22"
            fontFamily={logo.font}
            fill="#fff"
            fontWeight="600"
          >
            {logo.name}
          </text>
        </>
      )}

      {logo.layout === "text" && (
        <text
          x="50%"
          y="50%"
          textAnchor="middle"
          fontSize="32"
          fontFamily={logo.font}
          fill="#fff"
          fontWeight="700"
        >
          {logo.name}
        </text>
      )}
    </svg>
  );
}

/* 🔥 ICON RENDERER (100+ icons support) */
function IconRenderer({ name, color }: any) {
  const Icon = (Icons as any)[name];

  if (!Icon) {
    const Fallback = Icons.Star;
    return <Fallback size={42} color={color} strokeWidth={2} />;
  }

  return <Icon size={42} color={color} strokeWidth={2} />;
}

function pickFont(styles: string[]) {
  if (!styles) return "Inter";

  if (styles.includes("Elegant")) return "serif";
  if (styles.includes("Bold")) return "Poppins";

function pickFont(styles: string[]) {

import { NextResponse } from "next/server";
import fs from "fs";
import path from "path";

export async function GET() {
  try {
    const iconsDir = path.join(process.cwd(), "public/icons");

    const categories = fs.readdirSync(iconsDir);

    const result: any = {};

    for (const category of categories) {
      const categoryPath = path.join(iconsDir, category);

      // فقط فولدرها رو بخون
      if (!fs.statSync(categoryPath).isDirectory()) continue;

      const files = fs
        .readdirSync(categoryPath)
        .filter((file) => file.endsWith(".svg"))
        .map((file) => file.replace(".svg", ""));

      result[category] = files;
    }

    return NextResponse.json(result);
  } catch (err) {
    console.error(err);
    return NextResponse.json({ error: "Failed to load icons" }, { status: 500 });
  }
}

function pickIcon(category: string, icons: any) {

function pickFont(styles: string[]) {
  if (!styles) return "Inter";

  if (styles.includes("Elegant")) return "serif";
  if (styles.includes("Bold")) return "Poppins";
  if (styles.includes("Playful")) return "Poppins";

function IconRenderer({ name }: any) {
  return (
    <g transform="scale(1.6)">
      <image
        href={`/icons/${name}.svg`}
        x="-20"
        y="-20"
        width="40"
        height="40"
        preserveAspectRatio="xMidYMid meet"
      />
    </g>
  );
}
